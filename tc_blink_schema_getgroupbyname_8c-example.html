{
static const char input[] =
"InsertOrder/1 ->\n"
" string Symbol,\n"
" string OrderId,\n"
" u32 Price,\n"
" u32 Quantity\n"
"\n"
"CancelOrder/2 ->\n"
" string OrderId\n"
"\n"
"OrderInserted/3 ->\n"
" string OrderId\n"
"\n"
"OrderCanceled/4 ->\n"
" string OrderId\n";
static uint8_t heap[2048U];
static struct blink_pool pool;
*user = (void *)BLINK_Schema_new(BLINK_Pool_init(&pool, heap, sizeof(heap)), input, sizeof(input));
return 0;
}
